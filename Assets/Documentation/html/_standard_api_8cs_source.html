62 private string GenerateParameters(Parameters parameters)
63 {
64 FieldInfo[] fields = parameters.GetType().GetFields();
65 string queryString = "?";
66 foreach (var field in fields)
67 {
68
69 if (field.FieldType == typeof(string))
70 {
71 string value = (string)field.GetValue(parameters);
72 if (!string.IsNullOrEmpty(value))
73 {
74 queryString += $"{field.Name}={UnityWebRequest.EscapeURL(value)}&";
75 }
76 }
77
78 if (field.FieldType == typeof(int))
79 {
80 int value = (int)field.GetValue(parameters);
81 if (value >= 0)
82 {
83 queryString += $"{field.Name}={value}&";
84 }
85 }
86
87 if (field.FieldType == typeof(bool?))
88 {
89 bool? value = (bool?)field.GetValue(parameters);
90 if (value != null)
91 {
92 queryString += $"{field.Name}={value}&";
93 }
94 }
95 }
96
97 return queryString;
