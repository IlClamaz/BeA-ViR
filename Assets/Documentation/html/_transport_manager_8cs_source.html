245 [ClientRpc]
246 private void OnSightSeeing(float valuePressed) // -1 is Prev, 1 is Next
247 {
248 var sightPoints = curEnv.SightPoints;
249 var curSightPoint = curEnv.CurSightPoint;
250 if (curEnv.Father != null && sightPoints.Count == 0)
251 {
252 sightPoints = curEnv.Father.SightPoints;
253 curSightPoint = curEnv.Father.CurSightPoint;
254 }
255
256 if (sightPoints != null && sightPoints.Count > 0 && !cooldown)
257 {
258 SendFadeCamera();
259 SendReset();
260
261 cooldown = true;
262 if (valuePressed < 0) //Travel to PREV
263 {
264 if (curSightPoint == 0) curSightPoint = sightPoints.Count;
265 curSightPoint--;
266 }
267 else //Travel to NEXT
268 {
269 curSightPoint++;
270 if (curSightPoint == sightPoints.Count) curSightPoint = 0;
271 }
272 Teleport(sightPoints[curSightPoint].transform.position, -sightPoints[curSightPoint].transform.forward);
273 Invoke(nameof(CooldownReset), fadingDuration + 0.1f);
274
275 if (curEnv.Father != null && curEnv.SightPoints.Count == 0)
276 curEnv.Father.CurSightPoint = curSightPoint;
277 else curEnv.CurSightPoint = curSightPoint;
278 }
279 }
