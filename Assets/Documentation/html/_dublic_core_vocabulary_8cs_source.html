34 public static Property[] FromRoot(JObject root, string term)
35 {
36 JArray array;
37 array = Helpers.TryGet<JArray>(root, term, supressError: true);

39 if (array == null)
40 {

42 return null;
43 }

45 {
46 Property[] p = new Property[array.Count];
47 int i = 0;
48 foreach (var entry in array)

52 }
