449 private async Task<byte[]> ShowMedia(string url)//, RawImage container)
450 {
451 var request = UnityEngine.Networking.UnityWebRequestTexture.GetTexture(url);
452
453 await request.SendWebRequest();
454
455 if (!request.isNetworkError && !request.isHttpError)
456 {
457 processedImages++;
458 Texture2D tex = UnityEngine.Networking.DownloadHandlerTexture.GetContent(request);
459 // Encode.
460 byte[] bytes = tex.EncodeToJPG();
461 return bytes;
462 }
463 else
464 {
465 processedImages = -1;
